# GitPush Mini 🚀  
The fastest way to create GitHub repositories and upload entire projects directly from your phone — in under 7 taps.

![Version](https://img.shields.io/badge/version-1.0.0-blue)
![Platform](https://img.shields.io/badge/platform-iOS%20%7C%20Android-lightgrey)
![License](https://img.shields.io/badge/license-MIT-green)
![Status](https://img.shields.io/badge/status-active-brightgreen)

## Overview
GitPush Mini lets you:
- Log in with GitHub OAuth
- Create new repositories instantly
- Upload ZIP files or folders
- Auto-extract and auto-commit files
- Push entire projects from your phone

📁 app/screens/UploadScreen.js

import React, { useState } from 'react';
import { View, Text, Button } from 'react-native';
import * as DocumentPicker from 'expo-document-picker';
import { extractZip } from '../../utils/zip';

export default function UploadScreen({ route, navigation }) {
  const { token } = route.params;
  const [files, setFiles] = useState([]);

  async function pickZip() {
    const result = await DocumentPicker.getDocumentAsync({ type: 'application/zip' });
    if (result.assets) {
      const extracted = await extractZip(result.assets[0].uri);
      setFiles(extracted);
    }
  }

  return (
    <View style={{ padding: 40 }}>
      <Text style={{ fontSize: 24, fontWeight: 'bold' }}>
        Upload Project ZIP
      </Text>

      <Button title="Choose ZIP File" onPress={pickZip} />

      {files.length > 0 && (
        <Button
          title="Continue"
          onPress={() => navigation.navigate('Repo', { token, files })}
        />
      )}
    </View>
  );
}

📁 api/createRepo.js

export default async function handler(req, res) {
  const { token, name } = req.body;

  const response = await fetch("https://api.github.com/user/repos", {
    method: "POST",
    headers: {
      Authorization: `token ${token}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify({ name, private: false })
  });

  const data = await response.json();
  res.status(200).json(data);
}

📁 api/uploadFile.js

export default async function handler(req, res) {
  const { token, repo, path, content } = req.body;

  const response = await fetch(
    `https://api.github.com/repos/${repo}/contents/${path}`,
    {
      method: "PUT",
      headers: {
        Authorization: `token ${token}`,
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        message: "Upload file",
        content: Buffer.from(content).toString("base64")
      })
    }
  );

  const data = await response.json();
  res.status(200).json(data);
}

📁 utils/auth.js

import * as AuthSession from 'expo-auth-session';

export async function loginWithGitHub() {
  const redirectUri = AuthSession.makeRedirectUri();
  const authUrl = `https://github.com/login/oauth/authorize?client_id=YOUR_CLIENT_ID&scope=repo&redirect_uri=${redirectUri}`;

  const result = await AuthSession.startAsync({ authUrl });

  return result.params?.code || null;
}

📁 utils/zip.js

import JSZip from 'jszip';
import * as FileSystem from 'expo-file-system';

export async function extractZip(uri) {
  const zipData = await FileSystem.readAsStringAsync(uri, {
    encoding: FileSystem.EncodingType.Base64
  });

  const zip = await JSZip.loadAsync(zipData, { base64: true });

  const files = [];

  for (const filename of Object.keys(zip.files)) {
    const file = zip.files[filename];
    if (!file.dir) {
      const content = await file.async("string");
      files.push({ path: filename, content });
    }
  }

  return files;
}

📁 utils/github.js

export async function createRepo(token, name) {
  const response = await fetch("https://api.github.com/user/repos", {
    method: "POST",
    headers: {
      Authorization: `token ${token}`,
      "Content-Type": "application/json"
    },
    body: JSON.stringify({ name })
  });

  return response.json();
}

export async function uploadFiles(token, repoName, files) {
  for (const file of files) {
    await fetch(
      `https://api.github.com/repos/YOUR_GITHUB_USERNAME/${repoName}/contents/${file.path}`,
      {
        method: "PUT",
        headers: {
          Authorization: `token ${token}`,
          "Content-Type": "application/json"
        },
        body: JSON.stringify({
          message: "Upload file",
          content: Buffer.from(file.content).toString("base64")
        })
      }
    );
  }
}
